row 1개 찾기

gorm은 1개 row를 가져오기 위해 First, Take, Last 메서드를 제공한다.

이 메서드들은 LIMIT 1이라는 조건을 기본 쿼리에 추가한다. 그리고 만약 결과가 없을 때 ErrRecordNotFound 라는 error를 반환한다.

// TestSelectFindOne
func TestSelectFindOne(t *testing.T) {
	db := MustGormMySQLGormExampleDBConnectionPool()
	MigrationEntities(db, &User{}, &Language{})

	err := NewDummyDataGenerator[User](20).GenP(db, func(idx int, c *User) {
		c.ID = 0
		c.Name = gofakeit.Name()
		c.Age = int64(gofakeit.Number(28, 100))
		c.Birthday = randomTimeXYearsAgo(int(c.Age))
		c.CompanyID = GetCompanyId()
		c.Salary = gofakeit.RandomInt([]int{10000, 20000, 30000, 40000, 50000, 60000, 70000})
	})
	if err != nil {
		t.Fatal(err)
	}

	// 주어진 조건에서 검색된 값들에서 pk로 정렬된 순서로 처음인 값 1개를 가져온다
	// SELECT * FROM users ORDER BY id LIMIT 1;
	firstResult := User{}
	result := db.First(&firstResult)
	if result.Error != nil {
		t.Fatal(result.Error)
	}

	_ = result.RowsAffected // 가져온 레코드 갯수, 지운 레코드 갯수, 업데이트된 레코드 갯수
	_ = result.Error        // returns error or nil

	// check error ErrRecordNotFound
	errors.Is(result.Error, gorm.ErrRecordNotFound)

	// 주어진 조건에서 특정 순서 상관 없이 값 1 개를 가져온다

	takeResult := User{}
	if tx := db.Take(&takeResult, 2); tx.Error != nil {
		// SELECT * FROM users LIMIT 1;
		t.Fatal(tx.Error)
	}

	// 주어진 조건에서 검색된 값들에서 pk로 정렬된 순서로 마지막 값 1 개를 가져온다

	lastResult := User{ID: 6}
	if tx := db.Last(&lastResult); tx.Error != nil {
		// SELECT * FROM users ORDER BY id DESC LIMIT 1;
		t.Fatal(tx.Error)
	}

	//db.First(), db.Last() 메서드는 PK로 정렬된 순서에서 값을 각각 첫번째 ,마지막 값을 가져온다.
	//이 메서드는 데이터 저장 목적 구조체(destination struct) 가 포인터로 전달되거나
	//db.Model() 메서드에 gorm Model이 지정되었을 때에만 동작한다.

	var user User

	//데이터 저장 목적 구조체가 포인터로 전달되었으므로 값이 저장된다.
	db.First(&user)
	// SELECT * FROM `users` ORDER BY `users`.`id` LIMIT 1

	//db.Model() 메서드로 지정했기 때문에 map으로 데이터를 받아올 수 있다.
	result1 := map[string]interface{}{}
	if tx := db.Model(&User{}).First(&result1); tx.Error != nil {
		// SELECT * FROM `users` ORDER BY `users`.`id` LIMIT 1
		t.Fatal(tx)
	}

	//테이블 이름을 지정하더라도 기준 모델이 없기 때문에 map으로 데이터를 받아올 수 없다.
	result2 := map[string]interface{}{}

	if tx := db.Table(User{}.TableName()).First(&result2); tx.Error != nil {
		//SELECT * FROM `user1` ORDER BY `user1`. LIMIT 1ㅋ
		if errors.Is(tx.Error, gorm.ErrModelValueRequired) {
			//모델이 필요하다는 오류가 발생한다.
			fmt.Printf("%s", tx.Error)
		}
	}

	result2_last := map[string]interface{}{}
	if tx := db.Table(User{}.TableName()).Last(&result2_last); tx.Error != nil {
		if errors.Is(tx.Error, gorm.ErrModelValueRequired) {
			//모델이 필요하다는 오류가 발생한다.
			fmt.Printf("%s", tx.Error)
		}
	}

	// Take,Find 메서드로는 모델 없이도 동작한다.
	result3 := map[string]interface{}{}
	if tx := db.Table(User{}.TableName()).Take(&result3); tx.Error != nil {
		t.Fatal(tx)
	}

	//Find는 여러개의 값을 가져올때 주로 사용하는 메서드로, 1개 값을 가져올 수 있지만
	//Table Full Scan을 사용해 전체 row를 검색한뒤 첫번째 row를 가져오므로 비효율적이고
	//DBMS 엔진에 따라 정렬 순서가 달라지므로 예측할 수 없는 결과(Nondeterministic)를 가져올 수 있다.
	result4 := map[string]interface{}{}
	if tx := db.Table(User{}.TableName()).Find(&result4); tx.Error != nil {
		t.Fatal(tx)
	}
	//SELECT * FROM `user`

	//아래 구조체처럼 PK가 정의되지 않은 구조체를 사용하면 결과는 첫번째 필드인 Code를 기준으로 정렬된 값을 가져온다.
	//type Language struct {
	//	Code string
	//	Name string
	//}

	err = NewDummyDataGenerator[Language](20).GenP(db, func(idx int, c *Language) {
		c.Code = gofakeit.Language()
		c.Name = gofakeit.Name()
	})
	if err != nil {
		t.Fatal(err)
	}

	if tx := db.First(&Language{}); tx.Error != nil {
		t.Fatal(tx.Error)
	}
	// SELECT * FROM `languages` ORDER BY `languages`.`code` LIMIT 1

	fmt.Println(takeResult)
}

func TestRetrievingObjectsWithPrimaryKey(t *testing.T) {
	var user User
	var users []User
	db := MustGormMySQLGormExampleDBConnectionPool()

	tx1 := db.First(&user, 10)
	// SELECT * FROM users WHERE id = 10;
	assert.NoError(t, tx1.Error)

	var user2 User

	//pk가 문자열일때는 SQL injection 을 주의해야 한다.
	tx2 := db.First(&user2, "10")
	// SELECT * FROM `user1` WHERE `user1`.`id` = '10' ORDER BY `user1`.`id` LIMIT 1
	assert.NoError(t, tx2.Error)

	//sql injection 사례
	var user3 User
	userInputID := "1=1; drop table users;"
	tx3 := db.Find(&user3, userInputID) //mariadb에서는 syntax error가 발생한다.
	// SELECT * FROM users WHERE id = 10;
	assert.Error(t, tx3.Error)

	//pk가 문자열일 때 아래와 같이 사용하면 SQL injection 에서 안전하다.
	type UserIDString struct {
		ID   string `gorm:"primarykey"`
		Name string `gorm:"size:24"`
	}

	MigrationEntities(db, &UserIDString{})
	useridstring := UserIDString{ID: "1b74413f-f3b8-409f-ac47-e8c062e3472a", Name: "username"}
	tx41 := db.Create(&useridstring)
	if tx41.Error != nil {
		t.Fatal(tx41.Error)
	}
	//useridstring2 := UserIDString{}
	tx4 := db.First(&useridstring, "id = ?", useridstring.ID)
	// SELECT * FROM users WHERE id = "1b74413f-f3b8-409f-ac47-e8c062e3472a";
	assert.NoError(t, tx4.Error)

	//여러개 pk로 검색하기(in 연산자 사용)
	//limit이 사용되지 않는다.
	tx5 := db.Find(&users, []int{1, 2, 3})
	// SELECT * FROM users WHERE id IN (1,2,3);
	assert.NoError(t, tx5.Error)

	//데이터 저장 목적 구조체에 pk값이 있으면 검색조건으로 사용된다.
	//pk가 아닌 다른 필드는 조건으로 사용되지 않는다.
	var user4 = User{ID: 10, Name: "234"}
	tx6 := db.First(&user4)
	// SELECT * FROM users WHERE id = 10;
	assert.NoError(t, tx6.Error)

	//DeletedAt 같은 타입인 필드를 가지면 쿼리가 데이터를 가져오는 것이 다른 방식으로 동작한다.
	//gorm.DeletedAt 은 go std인 database/sql 패키지의 Valuer 및 Scanner 인터페이스를 구현했기 때문이다.
	//Value 메서드는 struct의 필드 값이 database로 serialize 될 때 변환을 수행하며
	//Scan 메서드는 database의 값을 read 할때 어떤 go type으로 변환될 것인지를 수행한다.
	type UserDeletedAt struct {
		ID        string         `gorm:"primarykey;size:16"`
		Name      string         `gorm:"size:24"`
		DeletedAt gorm.DeletedAt `gorm:"index"`
	}
	//

	MigrationEntities(db, &UserDeletedAt{})
	ids := NewIntSlice(1, 20, 1)
	err := NewDummyDataGenerator[UserDeletedAt](20).GenP(db, func(idx int, c *UserDeletedAt) {
		c.ID = strconv.Itoa(ids[idx])
		c.Name = gofakeit.Name()
		c.DeletedAt = gorm.DeletedAt{Valid: false}
	})
	if err != nil {
		t.Fatal(err)
	}

	var user5 = UserDeletedAt{ID: strconv.Itoa(15)}
	tableName, err := GormTableName(user5)
	if err != nil {
		t.Fatal(err)
	}
	tx7 := db.Table(tableName).First(&user5)
	// SELECT * FROM `users` WHERE `users`.`id` = '15' AND `users`.`deleted_at` IS NULL ORDER BY `users`.`id` LIMIT 1
	assert.NoError(t, tx7.Error)

	fmt.Println("")
}

func TestWhereStringCondition(t *testing.T) {
	db := MustGormMySQLGormExampleDBConnectionPool()
	MigrationEntities(db, &CreditCard{}, &Email{}, &User{})
	InitCompanyPicker()

	var searchword1 = "jinzhu"
	var searchword2 = "jinzhu2"
	err := CreateTestData()
	CheckFatal(t, err)

	err = db.Transaction(func(tx *gorm.DB) error {
		var user1 []User
		// Get first matched record
		if tx := tx.Where("name = ?", searchword1).Find(&user1); tx.Error != nil {
			return tx.Error
		}
		// SELECT * FROM users WHERE name = 'jinzhu' ORDER BY id LIMIT 1;

		// Get all matched records
		var allMatchedResult []User
		if tx := tx.Where("name <> ?", searchword2).Find(&allMatchedResult); tx.Error != nil {
			return tx.Error
		}
		// SELECT * FROM users WHERE name <> 'jinzhu';

		// IN
		var inResult []User
		if tx := tx.Where("name IN ?", []string{"jinzhu", "jinzhu2"}).Find(&inResult); tx.Error != nil {
			return tx.Error
		}
		// SELECT * FROM users WHERE name IN ('jinzhu','jinzhu 2');

		// LIKE
		var likeResult []User
		if tx := tx.Where("name LIKE ?", "%jin%").Find(&likeResult); tx.Error != nil {
			return tx.Error
		}
		// SELECT * FROM users WHERE name LIKE '%jin%';

		// AND
		var andResult []User
		if tx := tx.Where("name  = ? AND age >= ?", "jinzhu", "22").Find(&andResult); tx.Error != nil {
			return tx.Error
		}
		// SELECT * FROM contents WHERE name = 'jinzhu' AND age >= 22;

		// Time
		var timeResult []User
		rtgen := NewRandTimeGenerator()
		//updatedAt := rtgen.FromNMonthAgoToNow(3)
		if tx := tx.Where("updated_at > ?", rtgen.FromNMonthAgoToNow(3)).Find(&timeResult); tx.Error != nil {
			return tx.Error
		}
		// SELECT * FROM users WHERE updated_at > '2000-01-01 00:00:00';

		// BETWEEN
		var betweenResult1 []User
		from := rtgen.FromNMonthAgoToNow(1)
		to := time.Now()
		if tx := tx.Where("created_at BETWEEN ? AND ?", from, to).Find(&betweenResult1); tx.Error != nil {
			return tx.Error
		}
		// SELECT * FROM users WHERE created_at BETWEEN '2000-01-01 00:00:00' AND '2000-01-08 00:00:00';

		return nil
	})

	if err != nil {
		t.Fatal(err)
	}

	//다음 쿼리와 같은 경우 pk가 and 조건으로 입력되어 값이 없는 경우가 있을 수 있으므로
	//pk속성을 nil로 set해야 한다.
	var user = User{ID: 10}
	tx := db.Where("id = ?", 20).First(&user)
	//SELECT * FROM users WHERE id = 10 and id = 20 ORDER BY id ASC LIMIT 1

	if errors.Is(tx.Error, gorm.ErrRecordNotFound) {
		fmt.Println(tx.Error)
	}

	fmt.Println("")
}

func TestWhereUsingSliceAndMap(t *testing.T) {
	db := MustGormMySQLGormExampleDBConnectionPool()
	MigrationEntities(db, &User{}, &CreditCard{}, &Email{})
	err := CreateTestData()
	CheckFatal(t, err)

	err = db.Transaction(func(tx *gorm.DB) error {
		var structConditionsResult []User
		//struct로 where조건에서 zero values(0, '', false 등)는 where 절로 해석하지 않는다.
		if tx := tx.
			Where(&User{Name: "jinzhu", Age: 22}).
			Find(&structConditionsResult); tx.Error != nil {
			return tx.Error
		}

		//zero values까지 포함하려면 where 조건에 map을 사용하면 된다.
		var mapConditionsResult []User
		if tx := tx.
			Where(map[string]interface{}{"name": "jinzhu", "age": 22}).
			Find(&mapConditionsResult); tx.Error != nil {
			return tx.Error
		}
		// SELECT * FROM users WHERE name = "jinzhu" AND age = 20;

		//pk 값배열을 전달하면 in 으로 검색한다.
		var pkListConditionResult []User
		if tx := tx.
			Where([]int64{20, 21, 22}).
			Find(&pkListConditionResult); tx.Error != nil {
			return tx.Error
		}
		// SELECT * FROM users WHERE id IN (20, 21, 22);

		//아래처럼, 첫번째 arg 구조체에 있는 값중 where 절에서 조건으로 사용할 값을 선택할 수 있다.
		//아래의 첫번째 arg 구조체에는 age값이 사용하도록 선택했으므로 zero value 값으로 0이 조건에 사용되었다.
		var users1 []User
		db.Where(&User{Name: "jinzhu"}, "name", "Age").Find(&users1)
		// SELECT * FROM users WHERE name = "jinzhu" AND age = 0;

		//name이 첫번째 arg 구조체에 값이 있으나 조건으로 지정하지 않아 사용되지 않는다.
		var users2 []User
		db.Where(&User{Name: "jinzhu"}, "Age").Find(&users2)
		// SELECT * FROM users WHERE age = 0;

		return nil
	})

	if err != nil {
		t.Fatal(err)
	}

	fmt.Println("")
}

func TestInlineCondition(t *testing.T) {
	db := MustGormMySQLGormExampleDBConnectionPool()
	MigrationEntities(db, &User{})
	err := CreateTestData()
	CheckFatal(t, err)

	err = db.Transaction(func(tx *gorm.DB) error {
		type UserIDString struct {
			ID   string `gorm:"primarykey"`
			Name string `gorm:"size:24"`
		}

		MigrationEntities(db, &UserIDString{})
		useridstring := UserIDString{ID: "string_primary_key", Name: "username"}
		tx0 := db.Create(&useridstring)
		if tx0.Error != nil {
			t.Fatal(tx0.Error)
		}

		// Get by primary key if it were a non-integer type
		tx1 := db.First(&useridstring, "id = ?", "string_primary_key")
		// SELECT * FROM users WHERE id = 'string_primary_key';

		if tx1.Error != nil {
			return tx1.Error
		}

		var users2 = User{}
		// Plain SQL
		tx2 := db.Find(&users2, "name = ?", "jinzhu")
		// SELECT * FROM users WHERE name = "jinzhu";

		if tx2.Error != nil {
			return tx2.Error
		}

		var users3 = User{}
		tx3 := db.Find(&users3, "name <> ? AND age > ?", "jinzhu", 20)
		// SELECT * FROM users WHERE name <> "jinzhu" AND age > 20;

		if tx3.Error != nil {
			return tx3.Error
		}

		// Struct
		var users4 = User{}
		tx4 := db.Find(&users4, User{Age: 20})
		// SELECT * FROM users WHERE age = 20;

		if tx4.Error != nil {
			return tx4.Error
		}

		// Map
		var users5 = User{}
		tx5 := db.Find(&users5, map[string]interface{}{"age": 20})
		// SELECT * FROM users WHERE age = 20;

		if tx5.Error != nil {
			return tx5.Error
		}

		return nil
	})

	if err != nil {
		t.Fatal(err)
	}

	fmt.Println("")
}

func TestSelectingSpecificFields(t *testing.T) {
	db := MustGormMySQLGormExampleDBConnectionPool()

	err := CreateTestData()
	CheckFatal(t, err)

	err = db.Transaction(func(tx *gorm.DB) error {
		users1 := []User{}
		db.Select("name", "age").Find(&users1)
		// SELECT name, age FROM users;

		users2 := []User{}
		db.Select([]string{"name", "age"}).Find(&users2)
		// SELECT name, age FROM users;

		rows, err := db.Table("users").Select("COALESCE(age,?) age, id", 42).Rows()
		// SELECT COALESCE(age,'42') FROM users;

		if err != nil {
			return err
		}

		defer rows.Close()
		users3 := []User{}
		for rows.Next() {
			var entity User
			err := db.ScanRows(rows, &entity)
			if err != nil {
				return err
			}
			users3 = append(users3, entity)
		}

		return nil
	})

	if err != nil {
		t.Fatal(err)
	}

	fmt.Println("")
}

limit, offset 사용하기

Limit 은 검색할 최대 레코드 수를 지정하는 것이다.

Offset 은 레코드를 찾을 때 건너뛸 레코드 수를 지정하는 것이다.

select_04_order_limit_test.go-02
func TestLimitAndOffset(t *testing.T) {
	db := MustGormMySQLGormExampleDBConnectionPool()

	err := CreateTestData()
	CheckFatal(t, err)
	err = db.Transaction(func(tx *gorm.DB) error {
		users1 := []User{}
		tx1 := db.Limit(3).Find(&users1)
		// SELECT * FROM users LIMIT 3;

		if tx1.Error != nil {
			return tx1.Error
		}

		users2 := []User{}
		// Cancel limit condition with -1
		tx2 := db.Limit(10).Find(&users1).Limit(-1).Find(&users2)
		// SELECT * FROM users LIMIT 10; (users1)
		// SELECT * FROM users; (users2)

		if tx2.Error != nil {
			return tx2.Error
		}

		users3 := []User{}
		tx3 := db.Offset(3).Find(&users3)
		// SELECT * FROM users OFFSET 3;
		fmt.Println(db.Dialector.Name())
		if tx3.Error != nil {
			fmt.Println(tx3.Error.Error())
			if db.Dialector.Name() == "mysql" && tx3.Error != nil {
				assert.Error(t, tx3.Error)
			} else {
				return tx3.Error
			}
		}

		users4 := []User{}
		tx4 := db.Limit(10).Offset(5).Find(&users4)
		// SELECT * FROM users OFFSET 5 LIMIT 10;

		if tx4.Error != nil {
			return tx4.Error
		}

		users5 := []User{}
		// Cancel offset condition with -1
		tx5 := db.Offset(10).Find(&users1).Offset(-1).Find(&users5)
		// SELECT * FROM users OFFSET 10; (users1)
		// SELECT * FROM users; (users2)

		//mariadb에서는 limit없이 offset만 사용할 수 없다.
		fmt.Println(tx5.Error)
		if tx5.Error != nil {
			assert.Error(t, tx5.Error)
			if db.Dialector.Name() == "mysql" && tx3.Error != nil {
				fmt.Println(tx3.Error.Error())

			} else {
				return tx5.Error
			}
		}

		return nil
	})

	if err != nil {
		t.Fatal(err)
	}

	fmt.Println("")
}
