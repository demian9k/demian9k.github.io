	user3 := User{Name: "Jinzhu", Age: 18, Birthday: time.Now()}
	result3 := db.Omit("Name", "Age", "CreatedAt").Create(&user3)
	// INSERT INTO `users` (`birthday`,`updated_at`) VALUES ("2020-01-01 00:00:00.000", "2020-07-04 11:05:21.775")
	CheckFatal(t, result3.Error)

	//Batch Insert
	var users4 = []User{{Name: "jinzhu1"}, {Name: "jinzhu2"}, {Name: "jinzhu3"}}

	// batch size 100
	result4 := db.CreateInBatches(users4, 100)
	CheckFatal(t, result4.Error)
}

	CheckFatal(t, tx.Error)
}

func TestCreateWithDefaultValue(t *testing.T) {
	db := MustGormMySQLGormExampleDBConnectionPool()
	//CheckFatal(t, err)

	type User4 struct {
		ID   int64
		Name string `gorm:"default:galeone"`
		Age  int64  `gorm:"default:18"`
	}

	//  0, '', false 같은 zero value는 default 필드라도 database 에 저장되지 않으므로
	// 포인터로 정의하거나 Scanner/Valuer를 구현한 필드로 null 이 저장되도록 할 수 있다.
	type User5 struct {
		gorm.Model
		Name   string
		Age    *int         `gorm:"default:18"`
		Active sql.NullBool `gorm:"default:true"`
	}

	// 참고 데이터베이스에 기본값 또는 virtual/generated 값이 있는 필드에 기본 태그를 설정해야 하며,
	// 마이그레이션할 때 기본값 정의를 건너뛰려면 예를 들어 default:(-)를 사용할 수 있습니다:
	type User6 struct {
		ID        string `gorm:"default:uuid_generate_v3()"` // db func
		FirstName string
		LastName  string
		Age       uint8
		FullName  string `gorm:"->;type:GENERATED ALWAYS AS (concat(firstname,' ',lastname));default:(-);"`
	}

	entities := []interface{}{
		&User4{}, &User5{},
	}

	err := Migration(db, entities)

	CheckFatal(t, err)

	user4 := User4{}

	// name, age가 default 값으로 입력된다.
	tx := db.Table("user4").Create(&user4)
	//INSERT INTO `user4` (`name`,`age`) VALUES ('jinzhu4',18) RETURNING `id`

	user5 := User5{
		Name: "jinzhu5",
	}

	//age, active가 default 값으로 입력된다.
	tx2 := db.Table("user5").Create(&user5)
	//INSERT INTO `user5` (`created_at`, `updated_at`, `deleted_at`, `name`, `age`, `active`)
	//VALUES ('2023-11-19 02:31:52.079', '2023-11-19 02:31:52.079', NULL, 'jinzhu5', 18, true)
	//RETURNING `id`

	CheckFatal(t, tx.Error)
	CheckFatal(t, tx2.Error)
}

func TestUpsert_OnConflict(t *testing.T) {
	db := MustGormMySQLGormExampleDBConnectionPool()

	MigrationEntities(db, &User6{})

	user := User6{
		Name: "ddddd",
	}

	//conflict 시 아무것도 하지 않음
	tx1 := db.Clauses(clause.OnConflict{DoNothing: true}).Create(&user)
	CheckFatal(t, tx1.Error)

	//INSERT INTO `user1` (`name`, `age`, `birthday`, `created_at`, `updated_at`, `role`, `company_id`)
	//VALUES ('ddddd', 0, '0000-00-00 00:00:00', '2023-11-19 03:08:31.788', '2023-11-19 03:08:31.788', '', NULL)
	//ON DUPLICATE KEY UPDATE `id`=`id`
	//RETURNING `id`

	// 여러개의 row를 insert하기 위해 slice로 전달
	users1 := []*User6{
		{
			Name: "Jinzhu", Age: 18, Birthday: time.Now(),
		},
		{
			Name: "Jackson", Age: 19, Birthday: time.Now(),
		},
	}

	// Update columns to default value on `id` conflict
	tx2 := db.Clauses(clause.OnConflict{
		Columns:   []clause.Column{{Name: "id"}},
		DoUpdates: clause.Assignments(map[string]interface{}{"role": "user"}),
	}).Create(&users1)
	// MERGE INTO "users" USING *** WHEN NOT MATCHED THEN INSERT *** WHEN MATCHED THEN UPDATE SET ***; SQL Server
	// INSERT INTO `users` *** ON DUPLICATE KEY UPDATE ***; MySQL

	//INSERT INTO `user1` (`name`, `age`, `birthday`, `created_at`, `updated_at`, `role`, `company_id`)
	//VALUES ('Jinzhu', 18, '2023-11-19 03:08:31.839', '2023-11-19 03:08:31.839', '2023-11-19 03:08:31.839', '', NULL),
	//	('Jackson', 19, '2023-11-19 03:08:31.839', '2023-11-19 03:08:31.839', '2023-11-19 03:08:31.839', '', NULL)
	//ON DUPLICATE KEY UPDATE `role`='user'
	//RETURNING `id`
	//
	CheckFatal(t, tx2.Error)

	users2 := []*User6{
		{
			Name: "Jinzhu4", Age: 55, Birthday: time.Now(),
		},
		{
			Name: "Jackson22", Age: 66, Birthday: time.Now(),
		},
	}

	// Use SQL expression
	tx3 := db.Clauses(clause.OnConflict{
		Columns:   []clause.Column{{Name: "id"}},
		DoUpdates: clause.Assignments(map[string]interface{}{"count": gorm.Expr("GREATEST(count, VALUES(count))")}),
	}).Create(&users2)
	// INSERT INTO `users` *** ON DUPLICATE KEY UPDATE `count`=GREATEST(count, VALUES(count));

	//INSERT INTO `user6` (`created_at`, `updated_at`, `deleted_at`, `name`, `age`, `birthday`, `count`)
	//VALUES ('2023-11-19 03:08:31.889', '2023-11-19 03:08:31.889', NULL, 'Jinzhu4', 55, '2023-11-19 03:08:31.888', 0),
	//	('2023-11-19 03:08:31.889', '2023-11-19 03:08:31.889', NULL, 'Jackson22', 66, '2023-11-19 03:08:31.888', 0)
	//ON DUPLICATE KEY UPDATE `count`=GREATEST(count, VALUES(count))
	//RETURNING `id`

	CheckFatal(t, tx3.Error)

	users3 := []*User6{
		{
			Name: "Jinzhu66", Age: 75, Birthday: time.Now(),
		},
		{
			Name: "Jackson77", Age: 80, Birthday: time.Now(),
		},
	}

	// Update columns to new value on `id` conflict
	tx4 := db.Clauses(clause.OnConflict{
		Columns:   []clause.Column{{Name: "id"}},
		DoUpdates: clause.AssignmentColumns([]string{"name", "age"}),
	}).Create(&users3)
	// MERGE INTO "users" USING *** WHEN NOT MATCHED THEN INSERT *** WHEN MATCHED THEN UPDATE SET "name"="excluded"."name"; SQL Server
	// INSERT INTO "users" *** ON CONFLICT ("id") DO UPDATE SET "name"="excluded"."name", "age"="excluded"."age"; PostgreSQL
	// INSERT INTO `users` *** ON DUPLICATE KEY UPDATE `name`=VALUES(name),`age`=VALUES(age); MySQL

	//INSERT INTO `user1` (`name`, `age`, `birthday`, `created_at`, `updated_at`, `role`, `company_id`)
	//VALUES ('Jinzhu66', 75, '2023-11-19 03:08:31.938', '2023-11-19 03:08:31.939', '2023-11-19 03:08:31.939', '', NULL),
	//	('Jackson77', 80, '2023-11-19 03:08:31.938', '2023-11-19 03:08:31.939', '2023-11-19 03:08:31.939', '', NULL)
	//ON DUPLICATE KEY UPDATE `name`=VALUES(`name`),

	}

	// Update all columns to new value on conflict except primary keys and those columns having default values from sql func
	tx5 := db.Clauses(clause.OnConflict{
		UpdateAll: true,
	}).Create(&users4)
	// INSERT INTO "users" *** ON CONFLICT ("id") DO UPDATE SET "name"="excluded"."name", "age"="excluded"."age", ...;
	// INSERT INTO `users` *** ON DUPLICATE KEY UPDATE `name`=VALUES(name),`age`=VALUES(age), ...; MySQL

	//INSERT INTO `user1` (`name`, `age`, `birthday`, `created_at`, `updated_at`, `role`, `company_id`)

func NewGormMySQLConnection(dsn string) (*gorm.DB, error) {
	newLogger := gormLogger.New(
		log.New(os.Stdout, "\r\n", log.LstdFlags), // io writer
		gormLogger.Config{
			SlowThreshold:             time.Millisecond, // Slow SQL threshold
			LogLevel:                  gormLogger.Info,  // Log level
			IgnoreRecordNotFoundError: true,             // Ignore ErrRecordNotFound error for logger
			ParameterizedQueries:      false,            // Don't include params in the SQL log
			Colorful:                  true,             // Disable color
		},
	)

	db, err := gorm.Open(mysql.New(mysql.Config{
		DSN: dsn, // data source name
		//DefaultStringSize:         256,    // default size for string fields
		//DisableDatetimePrecision:  true,   // disable datetime precision, which not supported before MySQL 5.6
		//DontSupportRenameIndex:    true,   // drop & create when rename index, rename index not supported before MySQL 5.7, MariaDB
		//DontSupportRenameColumn:   true,   // `change` when rename column, rename column not supported before MySQL 8, MariaDB
		//SkipInitializeWithVersion: false,  // auto configure based on currently MySQL version
	}), &gorm.Config{
		Logger: newLogger,
	})
	if err != nil {
		return nil, errors.Wrap(err, "cannot connect to MySQL")
	}

	return db, nil
}

func SaveRowsAsSlice[T any](rows *sql.Rows, db *gorm.DB) ([]*T, error) {
	resultSlice := []*T{}

	var err error

	for rows.Next() {
		var entity T
		err = db.ScanRows(rows, &entity)
		if err != nil {
			break
		}
		resultSlice = append(resultSlice, &entity)
	}

	if err != nil {
		return nil, err
	}

	return resultSlice, nil
}

	if step <= 0 || end < start {
		return []int{}
	}
	s := make([]int, 0, 1+(end-start)/step)
	for start <= end {
		s = append(s, start)
		start += step
	}
	return s
}

func randomTimeXYearsAgo(X int) time.Time {
	// Get the current time
	currentTime := time.Now()

	// Determine the start and end of the target year
	startOfYear := time.Date(currentTime.Year()-X, 1, 1, 0, 0, 0, 0, currentTime.Location())
	endOfYear := time.Date(currentTime.Year()-X+1, 1, 1, 0, 0, 0, 0, currentTime.Location()).Add(-time.Nanosecond)

	// Calculate the number of seconds in that year
	secondsInYear := endOfYear.Sub(startOfYear).Seconds()

	// Generate a random number of seconds within that year
	randomSeconds := rand.Float64() * secondsInYear

	// Add the random number of seconds to the start of the target year
	randomTime := startOfYear.Add(time.Second * time.Duration(randomSeconds))

	return randomTime
}

func Migration(db *gorm.DB, entities []interface{}) error {
	err := db.Transaction(func(tx *gorm.DB) error {
		if err := db.Migrator().DropTable(entities...); err != nil {
			return err
		}

		if err := db.AutoMigrate(entities...); err != nil {
			return err
		}
		return nil
	})

	if err != nil {
		return err
	}
	return nil
}

	err := Migration(db, entities)
	if err != nil {
		panic(err)
	}

type RandTimeGenerator struct {
	random *rand.Rand
}

func NewRandTimeGenerator() RandTimeGenerator {
	return RandTimeGenerator{
		random: rand.New(rand.NewSource(rand.Int63())),
	}
}

func (rt RandTimeGenerator) RandomTimeBetween(start time.Time, end time.Time) time.Time {
	int64Range := Int64Range{min: start.Unix(), max: end.Unix()}
	randValue := int64Range.NextRandom(rt.random)
	return time.Unix(randValue, 0)
}

func (rt RandTimeGenerator) FromNDayAgoToNow(dayago int) time.Time {
	now := time.Now()
	startTime := DayAgo(dayago)
	return rt.RandomTimeBetween(startTime, now)
}

	now := time.Now()

	return rt.RandomTimeBetween(startTime, now)
}

	now := time.Now()

	return rt.RandomTimeBetween(startTime, now)

const Day = time.Hour * 24
const Month = Day * 30
const Year = Day * 365

func YearPeriod(year int) (time.Time, time.Time) {
	now := time.Now()
	start := now.Add(time.Duration(-year) * Year)
	return start, now
}
func DayAgo(day int) time.Time {
	dayago := time.Now().Add(time.Duration(-day) * Day)
	return dayago
}

func MonthAgo(month int) time.Time {
	monthago := time.Now().Add(time.Duration(-month) * Month)
	return monthago
}

func YearAgo(year int) time.Time {
	yearago := time.Now().Add(time.Duration(-year) * Year)
	return yearago
}

type Int64Range struct {
	min, max int64
}

func (ir *Int64Range) NextRandom(r *rand.Rand) int64 {
	return r.Int63n(ir.max-ir.min+1) + ir.min
}
