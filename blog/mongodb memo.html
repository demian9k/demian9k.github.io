bson 파일 import

docker exec -it testmongodb bash

#to import a .bson file:
mongorestore -d db_name -c collection_name /path/file.bson

# Incase only for a single collection.Try this:
mongorestore --drop -d db_name -c collection_name /path/file.bson

# For restoring the complete folder exported by mongodump:
mongorestore -d db_name /path/

#If you have enabled authentication use the below syntax:
mongorestore -u username --authenticationDatabase admin -d db_name -c collection_name /path/file.bson

mongorestore -u username -p mypasswd --authenticationDatabase admin -d db_name /path/

# mongorestore --host localhost --port 27017 --db db_name --authenticationDatabase db_name /mongodb_dump

// height가 170인 사람과 185인 직원을 조회합니다.
db.employee.find({ height: { $in: [170, 185] } })
