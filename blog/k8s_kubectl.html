# 노드 정보 자세히 보기
kubectl get nodes -o yaml


# 노드 정보에서 IP 보기
kubectl get nodes -o wide | awk -v OFS='\t\t' '{print $1, $6, $7}'


#ingress-controller 설치 후 nodePort 출력해서 환경변수로 값 할당하기
export IngHttp=$(kubectl get service -n ingress-nginx ingress-nginx-controller -o jsonpath='{.spec.ports[0].nodePort}')

EOF 로 inline에서 파일 입력하기

kubectl apply -f - <<EOF
apiVersion: v1
kind: Pod
metadata:
  name: busybox-sleep
spec:
  containers:
  - name: busybox
    image: busybox:1.28
    args:
    - sleep
    - "1000000"
---
apiVersion: v1
kind: Pod
metadata:
  name: busybox-sleep-less
spec:
  containers:
  - name: busybox
    image: busybox:1.28
    args:
    - sleep
    - "1000"
EOF

컨테이너 접속

kubectl exec --stdin --tty shell-demo -- /bin/bash

kubectl exec -it shell-demo /bin/bash

클러스터 내부에서 pod 하나를 만들고 다른 pod에 서비스를 통해 접속

# example-etcd-cluster-clien는 서비스명이고 2379는 해당서비스가 expose한 포트 번호

# etcdctl 실행하여 etcdcluster로 접속하기 위한 클라이언트 POD 실행
kubectl run --rm -i --tty etcdctl --image quay.io/coreos/etcd --restart=Never -- /bin/bash

export ETCDSVC=http://example-etcd-cluster-client:2379
#etcd에서 key로 값 가져오기
ectdctl --endpotins $ETCDSVC get foo

patch

# EtcdCluster 리소스(CR) 인스턴스인 example-etcd-cluster에서 spec.version을 3.3.12로 replace(op) 한다.
kubectl patch etcdcluster example-etcd-cluster --type 'json' -p '[{"op": "replace", "path": "/spec/version", "value":3.3.12}]'
